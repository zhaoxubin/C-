void LiftRevolve1(char *array,int k)
{
	char temp = *array;
	
	for (k ; k < 4; k++)
	{
		*(array +( k - 1)) = *(array + k);

	}
	*(array + 3) = temp;
	for (k = 0; k < 4; k++)
{
	printf("%c", *(array + k));
}

}

void LiftRevolve2(char *array,int k)
{
	int temp1 =*array;
	int temp2 = *(array + 1);
	*array = *(array + k);
	*(array + k - 1) = *(array + k + 1);
	*(array + k) = temp1;
	*(array + k + 1) = temp2;
	for (k = 0; k < 4; k++)
	{
		printf("%c", *(array + k));
	}
}
void LiftRevolve3(char *array,int k)
{
	char temp = *(array+k);

	for (k=3; k >0; k--)
	{
		*(array + k) = *((array + k-1));

	}
	*array = temp;
	for (k = 0; k < 4; k++)
	{
		printf("%c", *(array + k));
	}
	
}
